const redis = require('redis');
const client = redis.createClient({
  host: 'localhost',
  port: 6379
});

client.on('error', (err) => console.error('Redis error:', err));

// Set cache with TTL
async function setCache(key, value, ttlSeconds = 3600) {
  await client.setEx(key, ttlSeconds, JSON.stringify(value));
}

// Get from cache
async function getCache(key) {
  const data = await client.get(key);
  return data ? JSON.parse(data) : null;
}

// Delete from cache
async function deleteCache(key) {
  await client.del(key);
}

// Cache-aside pattern
async function getUserById(userId) {
  const cacheKey = `user:${userId}`;
  
  // Try cache first
  let user = await getCache(cacheKey);
  
  if (!user) {
    // Cache miss - fetch from database
    user = await db.query('SELECT * FROM users WHERE id = $1', [userId]);
    
    // Store in cache for 1 hour
    await setCache(cacheKey, user, 3600);
  }
  
  return user;
}
🏦 Banking: Account Balance Caching
▼ 
const express = require('express');
const app = express();

// Get account balance with caching
app.get('/api/accounts/:id/balance', async (req, res) => {
  const accountId = req.params.id;
  const cacheKey = `balance:${accountId}`;
  
  try {
    // Check cache
    let balance = await getCache(cacheKey);
    
    if (balance === null) {
      // Cache miss - query database
      const result = await db.query(
        'SELECT balance FROM accounts WHERE id = $1',
        [accountId]
      );
      
      balance = result.rows[0].balance;
      
      // Cache for 5 minutes
      await setCache(cacheKey, balance, 300);
    }
    
    res.json({ accountId, balance, cached: balance !== null });
  } catch (error) {
    res.status(500).json({ error: error.message });
  }
});

// Update balance and invalidate cache
app.post('/api/accounts/:id/deposit', async (req, res) => {
  const { id } = req.params;
  const { amount } = req.body;
  
  try {
    // Update database
    await db.query(
      'UPDATE accounts SET balance = balance + $1 WHERE id = $2',
      [amount, id]
    );
    
    // Invalidate cache
    await deleteCache(`balance:${id}`);
    
    res.json({ success: true });
  } catch (error) {
    res.status(500).json({ error: error.message });
  }
});

app.listen(3000);

🔑 Key Takeaways

  • Redis: Distributed in-memory cache, supports data structures
  • Cache-aside: App checks cache first, loads from DB on miss
  • TTL: Set expiration time to prevent stale data
  • Invalidation: Clear cache when data changes
  • Performance: Reduces DB load by 70-90%

✅ DO's

  • Set appropriate TTL for each data type
  • Invalidate cache on data updates
  • Handle cache failures gracefully
  • Monitor cache hit rate (aim for 80%+)
  • Use Redis for distributed caching

❌ DON'Ts

  • Don't cache sensitive data without encryption
  • Don't set TTL too long (stale data risk)
  • Don't cache everything (memory limits)
  • Don't forget to handle cache miss scenarios
  • Don't ignore cache server failures

const redis = require('redis');
const client = redis.createClient();

// Pattern 1: TTL-based expiration
async function cacheWithTTL(key, value, seconds) {
  await client.setEx(key, seconds, JSON.stringify(value));
}

// Pattern 2: Delete on update
async function updateUser(userId, updates) {
  // Update database
  await db.query('UPDATE users SET ... WHERE id = $1', [userId]);
  
  // Invalidate cache
  await client.del(`user:${userId}`);
}

// Pattern 3: Pattern-based deletion
async function invalidateUserCaches(userId) {
  const pattern = `user:${userId}:*`;
  const keys = await client.keys(pattern);
  
  if (keys.length > 0) {
    await client.del(keys);
  }
}

// Pattern 4: Write-through cache
async function saveUserWriteThrough(userId, data) {
  // Update database
  await db.query('UPDATE users SET data = $1 WHERE id = $2', [data, userId]);
  
  // Update cache immediately
  await client.setEx(`user:${userId}`, 3600, JSON.stringify(data));
}

// Pattern 5: Lazy invalidation (refresh on read)
async function getWithRefresh(key, fetchFunc, ttl = 3600) {
  let data = await client.get(key);
  
  if (!data) {
    data = await fetchFunc();
    await client.setEx(key, ttl, JSON.stringify(data));
  }
  
  return JSON.parse(data);
}

    throw error;
  } finally {
    dbClient.release();
  }
}

// Cache manager with versioning
class CacheManager {
  async set(key, value, ttl = 3600) {
    const version = Date.now();
    const data = { value, version };
    await client.setEx(key, ttl, JSON.stringify(data));
  }
  
  async get(key) {
    const data = await client.get(key);
    if (!data) return null;
    
    const { value, version } = JSON.parse(data);
    return { value, version };
  }
  
  async invalidate(pattern) {
    const keys = await client.keys(pattern);
    if (keys.length > 0) {
      await client.del(keys);
    }
  }
}

🔑 Key Takeaways

  • Two hard problems: Cache invalidation, naming things, off-by-one errors
  • TTL approach: Simple but may serve stale data
  • Write-through: Updates cache immediately on data change
  • Event-driven: Invalidate cache when data changes
  • Pattern matching: Use Redis KEYS or SCAN for bulk invalidation

✅ DO's

  • Invalidate related caches together
  • Use event-driven invalidation for consistency
  • Set conservative TTLs for critical data
  • Monitor cache invalidation frequency
  • Use versioning for cache entries

❌ DON'Ts

  • Don't rely only on TTL for financial data
  • Don't use KEYS in production (blocks Redis)
  • Don't forget cascade invalidation
  • Don't invalidate too frequently (defeats caching)
  • Don't ignore invalidation errors

// accountService.js
class AccountService {
  constructor(db) {
    this.db = db;
  }
  
  async getBalance(accountId) {
    const account = await this.db.findAccount(accountId);
    return account.balance;
  }
  
  async deposit(accountId, amount) {
    if (amount <= 0) {
      throw new Error('Amount must be positive');
    }
    
    const account = await this.db.findAccount(accountId);
    account.balance += amount;
    await this.db.saveAccount(account);
    
    return account.balance;
  }
}

module.exports = AccountService;

// accountService.test.js
const AccountService = require('./accountService');

describe('AccountService', () => {
  let accountService;
  let mockDb;
  
  beforeEach(() => {
    // Arrange: Setup mock database
    mockDb = {
      findAccount: jest.fn(),
      saveAccount: jest.fn()
    };
    
    accountService = new AccountService(mockDb);
  });
  
  describe('getBalance', () => {
    test('returns account balance', async () => {
      // Arrange
      mockDb.findAccount.mockResolvedValue({ 
        id: 1, 
        balance: 1000 
      });
      
      // Act
      const balance = await accountService.getBalance(1);
      
      // Assert
      expect(balance).toBe(1000);
      expect(mockDb.findAccount).toHaveBeenCalledWith(1);
    });
  });
  
  describe('deposit', () => {
    test('increases balance correctly', async () => {
      // Arrange
      const mockAccount = { id: 1, balance: 1000 };
      mockDb.findAccount.mockResolvedValue(mockAccount);
      mockDb.saveAccount.mockResolvedValue(true);
      
      // Act
      const newBalance = await accountService.deposit(1, 500);
      
      // Assert
      expect(newBalance).toBe(1500);
      expect(mockDb.saveAccount).toHaveBeenCalledWith(mockAccount);
    });
    
    test('throws error for negative amount', async () => {
      await expect(
        accountService.deposit(1, -100)
      ).rejects.toThrow('Amount must be positive');
    });
  });
});

🔑 Key Takeaways

  • Jest: Zero-config, includes assertions, mocking, coverage
  • AAA pattern: Arrange (setup), Act (execute), Assert (verify)
  • Mocking: Use jest.fn() for function mocks, jest.mock() for modules
  • Coverage: Aim for 80%+ coverage, run with --coverage flag
  • Test isolation: Each test should be independent

✅ DO's

  • Write descriptive test names
  • Use beforeEach for test setup
  • Mock external dependencies
  • Test edge cases and errors
  • Keep tests simple and focused

❌ DON'Ts

  • Don't test implementation details
  • Don't share state between tests
  • Don't make tests dependent on order
  • Don't test external APIs directly
  • Don't ignore failing tests

app.post('/api/auth/login', async (req, res) => {
  const { username, password } = req.body;
  
  // Verify credentials
  const user = await db.findUser(username);
  const valid = await bcrypt.compare(password, user.passwordHash);
  
  if (!valid) {
    return res.status(401).json({ error: 'Invalid credentials' });
  }
  
  const payload = { userId: user.id, role: user.role };
  
  // Generate tokens
  const accessToken = generateAccessToken(payload);
  const refreshToken = generateRefreshToken(payload);
  
  // Store refresh token in Redis
  await client.setEx(

    refreshToken
  );
  
  res.json({ accessToken, refreshToken });
});

app.post('/api/auth/refresh', async (req, res) => {
  const { refreshToken } = req.body;
  
  if (!refreshToken) {
    return res.status(401).json({ error: 'Refresh token required' });
  }
  
  try {
    // Verify refresh token
    const decoded = jwt.verify(refreshToken, process.env.REFRESH_SECRET);
    
    // Check if token exists in Redis
    const stored = await client.get(`refresh:${decoded.userId}`);
    
    if (stored !== refreshToken) {
      return res.status(403).json({ error: 'Invalid refresh token' });
    }
    
    // Generate new access token
    const accessToken = generateAccessToken({
      userId: decoded.userId,
      role: decoded.role
    });
    
    res.json({ accessToken });
  } catch (error) {
    res.status(403).json({ error: 'Invalid refresh token' });
  }
});

    const decoded = jwt.verify(refreshToken, process.env.REFRESH_SECRET);
    
    // Remove refresh token from Redis

    res.json({ message: 'Logged out successfully' });

async function authenticateToken(req, res, next) {
  const authHeader = req.headers['authorization'];
  const token = authHeader && authHeader.split(' ')[1];

    return res.status(401).json({ error: 'Access token required' });

}

  const state = crypto.randomBytes(16).toString('hex');
  
  // Store state in session
  req.session.oauthState = state;
  
  const authUrl = `${OAUTH_CONFIG.authorizationUrl}?` +
    `response_type=code&` +
    `client_id=${OAUTH_CONFIG.clientId}&` +
    `redirect_uri=${OAUTH_CONFIG.redirectUri}&` +
    `scope=email profile&` +
    `state=${state}`;
  
  res.redirect(authUrl);
});

// Step 2: Handle callback with authorization code
app.get('/callback', async (req, res) => {
  const { code, state } = req.query;
  
  // Verify state to prevent CSRF
  if (state !== req.session.oauthState) {
    return res.status(400).json({ error: 'Invalid state' });
  }
  
  try {
    // Exchange code for access token
    const tokenResponse = await axios.post(OAUTH_CONFIG.tokenUrl, {
      grant_type: 'authorization_code',
      code,
      redirect_uri: OAUTH_CONFIG.redirectUri,
      client_id: OAUTH_CONFIG.clientId,
      client_secret: OAUTH_CONFIG.clientSecret
    });
    
    const { access_token, refresh_token, expires_in } = tokenResponse.data;
    
    // Get user info
    const userResponse = await axios.get(
      'https://oauth-provider.com/userinfo',
      {
        headers: { Authorization: `Bearer ${access_token}` }
      }
    );
    
    // Store tokens securely
    req.session.accessToken = access_token;
    req.session.refreshToken = refresh_token;
    req.session.user = userResponse.data;
    
    res.redirect('/dashboard');
  } catch (error) {
    res.status(500).json({ error: 'OAuth failed' });
  }
});
