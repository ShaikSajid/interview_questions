async function readTransactionLog() {

async function saveConfig(config) {
  const tempFile = 'config.json.tmp';
  const finalFile = 'config.json';
  
  await writeFile(tempFile, JSON.stringify(config, null, 2));
  await fs.rename(tempFile, finalFile); // Atomic operation
}

// Create directory structure

  const logDir = join(process.cwd(), 'logs');
  await mkdir(logDir, { recursive: true });
}
🏦 Banking: Audit Log Management
▼ 
import { appendFile, readdir, stat, unlink } from 'fs/promises';
import { createWriteStream } from 'fs';
import { join } from 'path';

class AuditLogger {
  constructor(logDir = './audit-logs') {
    this.logDir = logDir;
    this.maxFileSize = 10 * 1024 * 1024; // 10MB
    this.maxAge = 90 * 24 * 60 * 60 * 1000; // 90 days
  }

  async log(event, data) {
    const timestamp = new Date().toISOString();
    const logEntry = JSON.stringify({
      timestamp,
      event,
      data,
      userId: data.userId,
      ipAddress: data.ipAddress
    });

    const filename = this.getCurrentLogFile();
    await appendFile(filename, logEntry + '\n');
    
    // Check if rotation needed
    await this.rotateIfNeeded(filename);
  }

  getCurrentLogFile() {
    const date = new Date().toISOString().split('T')[0];
    return join(this.logDir, `audit-${date}.log`);
  }

  async rotateIfNeeded(filename) {
    const stats = await stat(filename);
    
    if (stats.size > this.maxFileSize) {
      const rotated = `${filename}.${Date.now()}`;
      await fs.rename(filename, rotated);
      console.log(`Rotated log file: ${rotated}`);
    }
  }

  // Clean up old logs (compliance: 90 days retention)
  async cleanup() {
    const files = await readdir(this.logDir);
    const now = Date.now();

    for (const file of files) {
      const filepath = join(this.logDir, file);
      const stats = await stat(filepath);
      
      if (now - stats.mtime.getTime() > this.maxAge) {
        await unlink(filepath);
        console.log(`Deleted old log: ${file}`);
      }
    }
  }
}

🔑 Key Takeaways

  • Use fs.promises for modern async/await file operations
  • Atomic writes prevent data corruption (write temp → rename)
  • File rotation prevents single files from growing too large
  • Always handle errors in file operations (disk full, permissions)
  • Use streams for large files (>100MB)

✅ DO's

  • Use fs.promises or util.promisify for async operations
  • Implement proper error handling and retries
  • Use path.join() for cross-platform paths
  • Implement log rotation for production systems
  • Set appropriate file permissions (0600 for sensitive data)

❌ DON'Ts

  • Don't use sync methods in production (blocks event loop)
  • Don't ignore ENOENT errors (file not found)
  • Don't hardcode file paths (use environment variables)
  • Don't forget to close file descriptors (use try-finally)
  • Don't write sensitive data without encryption

import http from 'http';
import url from 'url';

const accounts = new Map();

const server = http.createServer(async (req, res) => {
  const parsedUrl = url.parse(req.url, true);
  const pathname = parsedUrl.pathname;
  const method = req.method;

  // GET /accounts - List all accounts
  if (method === 'GET' && pathname === '/accounts') {
    res.writeHead(200, { 'Content-Type': 'application/json' });
    res.end(JSON.stringify(Array.from(accounts.values())));
  }

  // POST /accounts - Create new account
  else if (method === 'POST' && pathname === '/accounts') {
    const body = await getRequestBody(req);
    const { name, balance } = JSON.parse(body);
    
    const id = 'ACC' + Date.now();
    const account = { id, name, balance, createdAt: new Date() };
    accounts.set(id, account);
    
    res.writeHead(201, { 'Content-Type': 'application/json' });
    res.end(JSON.stringify(account));
  }

  // PUT /accounts/:id - Update account
  else if (method === 'PUT' && pathname.match(/^\/accounts\/[\w-]+$/)) {
    const id = pathname.split('/')[2];
    const body = await getRequestBody(req);
    const updates = JSON.parse(body);
    
    if (accounts.has(id)) {
      const account = { ...accounts.get(id), ...updates };
      accounts.set(id, account);
      res.writeHead(200, { 'Content-Type': 'application/json' });
      res.end(JSON.stringify(account));
    } else {
      res.writeHead(404, { 'Content-Type': 'application/json' });
      res.end(JSON.stringify({ error: 'Account not found' }));
    }
  }

  // DELETE /accounts/:id - Delete account
  else if (method === 'DELETE' && pathname.match(/^\/accounts\/[\w-]+$/)) {
    const id = pathname.split('/')[2];
    
    if (accounts.delete(id)) {
      res.writeHead(204);
      res.end();
    } else {
      res.writeHead(404);
      res.end(JSON.stringify({ error: 'Account not found' }));
    }
  }
});

function getRequestBody(req) {
  return new Promise((resolve) => {
    let body = '';
    req.on('data', chunk => body += chunk);
    req.on('end', () => resolve(body));
  });
}

import express from 'express';

const app = express();
app.use(express.json());

const transactions = [];

// GET /transactions - Retrieve transaction history
app.get('/api/transactions', (req, res) => {
  const { accountId, limit = 10 } = req.query;
  
  let filtered = transactions;
  if (accountId) {
    filtered = transactions.filter(t => t.accountId === accountId);
  }
  
  res.json({
    total: filtered.length,
    data: filtered.slice(0, parseInt(limit))
  });
});

// POST /transactions - Create new transaction
app.post('/api/transactions', (req, res) => {
  const { accountId, amount, type } = req.body;
  
  if (!accountId || !amount || !type) {
    return res.status(400).json({ error: 'Missing required fields' });
  }
  
  const transaction = {
    id: 'TXN' + Date.now(),
    accountId,
    amount,
    type,
    status: 'pending',
    timestamp: new Date()
  };
  
  transactions.push(transaction);
  res.status(201).json(transaction);
});

// PUT /transactions/:id - Update transaction status
app.put('/api/transactions/:id', (req, res) => {
  const { id } = req.params;
  const { status } = req.body;
  
  const transaction = transactions.find(t => t.id === id);
  
  if (!transaction) {
    return res.status(404).json({ error: 'Transaction not found' });
  }
  
  transaction.status = status;
  res.json(transaction);
});

// DELETE /transactions/:id - Cancel transaction
app.delete('/api/transactions/:id', (req, res) => {
  const { id } = req.params;
  const index = transactions.findIndex(t => t.id === id);
  
  if (index === -1) {
    return res.status(404).json({ error: 'Transaction not found' });
  }
  
  transactions.splice(index, 1);
  res.status(204).send();
});

app.listen(3000);

🔑 Key Takeaways

  • Idempotency: GET, PUT, DELETE are safe to retry
  • Status codes: 200 (OK), 201 (Created), 204 (No Content), 404 (Not Found)
  • RESTful design: Use nouns for resources, verbs for actions
  • POST vs PUT: POST creates, PUT updates/replaces

✅ DO's

  • Use appropriate HTTP methods for CRUD operations
  • Return correct status codes (2xx, 4xx, 5xx)
  • Make GET, PUT, DELETE idempotent
  • Use Content-Type headers correctly
  • Version your APIs (/v1/accounts)

❌ DON'Ts

  • Don't use GET for operations that modify data
  • Don't ignore request body parsing errors
  • Don't return 200 for errors (use 4xx/5xx)
  • Don't forget to validate input data
  • Don't expose internal error details to clients

  }
};

class InterestScheduler {
  constructor() {
    this.intervalId = null;
    this.accounts = new Map();
  }

  // Calculate interest monthly
  start() {
    console.log('Interest scheduler started');
    
    // Run every hour (in production: every 24 hours)
    this.intervalId = setInterval(() => {
      this.calculateInterest();
    }, 60 * 60 * 1000); // 1 hour
    
    // Run immediately on start
    this.calculateInterest();
  }

  calculateInterest() {
    const rate = 0.05 / 12; // 5% annual / 12 months
    
    for (const [id, account] of this.accounts) {
      const interest = account.balance * rate;
      account.balance += interest;
      
      console.log(`Account ${id}: Interest $${interest.toFixed(2)}`);
      console.log(`New Balance: $${account.balance.toFixed(2)}`);
    }
  }

  stop() {
    if (this.intervalId) {
      clearInterval(this.intervalId);
      console.log('Interest scheduler stopped');
    }
  }

  addAccount(id, balance) {
    this.accounts.set(id, { balance });
  }
}

// Session timeout handler
class SessionManager {
  constructor(timeoutMs = 15 * 60 * 1000) {
    this.timeoutMs = timeoutMs;
    this.sessions = new Map();
  }

  createSession(userId) {
    const sessionId = 'SES' + Date.now();
    
    const timeoutId = setTimeout(() => {
      this.expireSession(sessionId);
    }, this.timeoutMs);
    
    this.sessions.set(sessionId, { userId, timeoutId });
    return sessionId;
  }

  refreshSession(sessionId) {
    const session = this.sessions.get(sessionId);
    
    if (session) {
      clearTimeout(session.timeoutId);
      session.timeoutId = setTimeout(() => {
        this.expireSession(sessionId);
      }, this.timeoutMs);
    }
  }

  expireSession(sessionId) {
    console.log(`Session ${sessionId} expired`);
    this.sessions.delete(sessionId);
  }
}

🔑 Key Takeaways

  • setInterval executes callback repeatedly at regular intervals
  • Timer precision is not guaranteed; delays may vary slightly
  • Always clean up timers with clearTimeout/clearInterval
  • Timers don't block the event loop; code continues execution

  • Store timer IDs to cancel them later
  • Use setTimeout for one-time delayed execution

  • Consider using cron jobs for scheduled tasks

  • Don't create intervals without cleanup
  • Don't use setInterval for recursive operations
  • Don't forget to handle errors in timer callbacks
  • Don't use timers for high-frequency operations

class TransactionProcessor {
  // High-priority (use nextTick)
  processUrgent(transaction) {
    process.nextTick(() => {
      this.validateTransaction(transaction);
      this.executeTransaction(transaction);
    });
  }

  // Normal-priority (use setImmediate)
  processNormal(transaction) {
    setImmediate(() => {
      this.validateTransaction(transaction);
      this.executeTransaction(transaction);
    });
  }

  // Batch processing (safe recursion)
  processBatch(transactions) {
    if (transactions.length === 0) return;
    
    const txn = transactions.shift();
    setImmediate(() => {
      this.executeTransaction(txn);
      this.processBatch(transactions);
    });
  }
}

🔑 Key Takeaways

  • process.nextTick has highest priority, executes before event loop continues
  • setImmediate executes in Check phase, after I/O events
  • Use nextTick for emitting events, error callbacks, cleanup
  • Use setImmediate for deferring work, recursive operations
  • Avoid nextTick recursion - can starve the event loop

✅ DO's

  • Use process.nextTick for event emissions
  • Use setImmediate for breaking up CPU-intensive work
  • Prefer setImmediate for recursive operations
  • Use nextTick to ensure callbacks run after construction
  • Use setImmediate when you need to yield to I/O

❌ DON'Ts

  • Don't use recursive nextTick (causes starvation)
  • Don't use nextTick for long-running operations
  • Don't confuse setTimeout(fn, 0) with setImmediate
  • Don't overuse nextTick (impacts performance)
  • Don't use nextTick when setImmediate is more appropriate

// SIGTERM - Graceful termination (from OS/Docker/K8s)
process.on('SIGTERM', () => {
  console.log('SIGTERM received, shutting down gracefully');
  server.close(() => {
    console.log('Server closed');
    process.exit(0);
  });
});

// SIGINT - Interrupt (Ctrl+C)
process.on('SIGINT', () => {
  console.log('\nSIGINT received (Ctrl+C)');
  gracefulShutdown('SIGINT');
});

// Custom signals
process.on('SIGUSR1', () => {
  console.log('SIGUSR1: Dumping memory stats');
  console.log(process.memoryUsage());
});

// Uncaught exceptions
process.on('uncaughtException', (error) => {
  console.error('Uncaught Exception:', error);
  process.exit(1);
});

// Unhandled promise rejections
process.on('unhandledRejection', (reason, promise) => {
  console.error('Unhandled Rejection at:', promise, 'reason:', reason);
  process.exit(1);
});
🏦 Banking: Production Graceful Shutdown

import express from 'express';
import { createPool } from 'mysql2/promise';

const app = express();
const dbPool = createPool({ /* config */ });

let isShuttingDown = false;
const activeTransactions = new Set();

// Track active transactions
app.use((req, res, next) => {
  if (isShuttingDown) {
    return res.status(503).send('Server shutting down');
  }
  
  const transactionId = `TXN-${Date.now()}`;
  activeTransactions.add(transactionId);
  
  res.on('finish', () => {
    activeTransactions.delete(transactionId);
  });
  
  next();
});

// Graceful shutdown handler
async function gracefulShutdown(signal) {
  console.log(`${signal} received, starting graceful shutdown`);
  isShuttingDown = true;

  // Step 1: Stop accepting new requests
  server.close(async () => {
    console.log('✓ HTTP server closed');

    // Step 2: Wait for active transactions
    console.log(`Waiting for ${activeTransactions.size} active transactions...`);
    while (activeTransactions.size > 0) {
      await new Promise(resolve => setTimeout(resolve, 100));
    }
    console.log('✓ All transactions completed');

    // Step 3: Close database connections
    await dbPool.end();
    console.log('✓ Database connections closed');

    // Step 4: Close other resources (Redis, queues, etc.)
    await closeRedis();
    await closeMessageQueue();
    console.log('✓ All resources closed');

    console.log('✓ Graceful shutdown complete');
    process.exit(0);
  });

  // Timeout: Force shutdown after 30 seconds
  setTimeout(() => {
    console.error('⚠️ Forced shutdown after timeout');
    process.exit(1);
  }, 30000);
}

// Register signal handlers
process.on('SIGTERM', () => gracefulShutdown('SIGTERM'));
process.on('SIGINT', () => gracefulShutdown('SIGINT'));

// Health check endpoint (for load balancer)
app.get('/health', (req, res) => {
  if (isShuttingDown) {
    return res.status(503).json({ status: 'shutting down' });
  }
  res.json({ status: 'ok' });
});

const server = app.listen(3000, () => {
  console.log('Banking API running on port 3000');
});

🔑 Key Takeaways

  • SIGTERM for graceful termination from Docker/K8s
  • SIGINT for keyboard interrupt (Ctrl+C)
  • Graceful shutdown: Stop accepting → Complete requests → Close connections
  • Timeout forced shutdown to prevent hanging
  • Health checks inform load balancers of shutdown status

✅ DO's

  • Handle both SIGTERM and SIGINT signals
  • Stop accepting new connections first
  • Wait for in-flight requests to complete
  • Close database connections gracefully
  • Implement timeout for forced shutdown (30s)

❌ DON'Ts

  • Don't use process.exit() immediately on signal
  • Don't leave database connections open
  • Don't forget to drain message queues
  • Don't ignore in-flight transactions
  • Don't skip health check endpoint updates
