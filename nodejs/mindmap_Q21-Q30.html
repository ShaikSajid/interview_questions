⚡ Cluster Module

Spawn worker processes to utilize all CPU cores

👥 Master/Worker Architecture

Master manages workers, workers handle requests

Round-robin distribution across workers

💡 Basic Cluster Implementation
▼ 
const cluster = require('cluster');
const http = require('http');
const os = require('os');

if (cluster.isMaster) {
  console.log(`Master process ${process.pid} starting`);
  
  const numCPUs = os.cpus().length;
  console.log(`Forking ${numCPUs} workers...`);
  
  // Fork workers
  for (let i = 0; i < numCPUs; i++) {
    cluster.fork();
  }
  
  // Handle worker exit
  cluster.on('exit', (worker, code, signal) => {
    console.log(`Worker ${worker.process.pid} died`);
    console.log('Starting a new worker...');
    cluster.fork();
  });
  
} else {
  // Workers share the same server port
  const server = http.createServer((req, res) => {
    res.writeHead(200);
    res.end(`Handled by worker ${process.pid}\n`);
  });
  
  server.listen(3000, () => {
    console.log(`Worker ${process.pid} started`);
  });
}

const cluster = require('cluster');

const os = require('os');

if (cluster.isMaster) {
  const numWorkers = os.cpus().length;
  console.log(`🏦 Banking API - Master starting with ${numWorkers} workers`);
  
  const workerMetrics = new Map();
  
  for (let i = 0; i < numWorkers; i++) {
    const worker = cluster.fork();
    workerMetrics.set(worker.id, { requests: 0, errors: 0 });
  }
  
  // Track worker metrics
  cluster.on('message', (worker, msg) => {
    if (msg.type === 'request') {
      const metrics = workerMetrics.get(worker.id);
      metrics.requests++;
    }
  });
  
  // Graceful restart
  cluster.on('exit', (worker, code, signal) => {
    const metrics = workerMetrics.get(worker.id);
    console.log(`Worker ${worker.id} died (handled ${metrics.requests} requests)`);
    
    workerMetrics.delete(worker.id);
    const newWorker = cluster.fork();
    workerMetrics.set(newWorker.id, { requests: 0, errors: 0 });
  });
  
  // Display metrics every 10 seconds
  setInterval(() => {
    let totalRequests = 0;
    workerMetrics.forEach(m => totalRequests += m.requests);

💡 Stack vs Heap Memory Allocation
▼ 
// Stack vs Heap Example

function processPayment(amount) {
  // STACK: Primitives stored directly on stack
  const fee = amount * 0.02;              // Stack
  const total = amount + fee;             // Stack
  const isValid = amount > 0;           // Stack
  
  // HEAP: Objects stored on heap, reference on stack
  const payment = {                       // Heap (object)
    amount: amount,
    fee: fee,
    total: total,
    timestamp: new Date()             // Heap (Date object)
  };
  
  const tags = ['urgent', 'verified'];    // Heap (array)
  
  return payment;  // Stack cleared, heap data persists until GC
}

// Memory Layout:
// STACK:          HEAP:
// fee   → 20      payment → { amount: 1000, fee: 20, ... }
// total → 1020    tags → ['urgent', 'verified']
// isValid → true  Date → Sat Nov 30 2025...
// payment → 0x1A2B (reference to heap)

// V8 Heap Structure
/*
 * New Space (Young Generation) ~1-8MB
 *   - Nursery: New allocations
 *   - Intermediate: Survived one GC
 *
 * Old Space (Old Generation) ~hundreds of MB
 *   - Old Pointer Space: Objects with pointers
 *   - Old Data Space: Objects without pointers
 *
 * Large Object Space
 *   - Objects > 1MB
 *
 * Code Space
 *   - JIT compiled code
 */

class TransactionProcessor {
  // ❌ BAD: Loads all transactions into memory
  async processAllTransactions_BAD() {
    // This loads 1 million transactions (~500MB) into heap!
    const transactions = await db.query('SELECT * FROM transactions');
    
    for (const txn of transactions) {
      await this.processSingle(txn);
    }
  }
  
  // ✅ GOOD: Stream-based processing (constant memory)
  async processAllTransactions_GOOD() {
    const stream = db.stream('SELECT * FROM transactions');
    
    for await (const txn of stream) {
      await this.processSingle(txn);
      // Each txn is GC'd after processing
    }
  }
  
  // Memory-efficient batch processing
  async processBatch(batchSize = 1000) {
    let offset = 0;
    let hasMore = true;
    
    while (hasMore) {
      // Process in chunks to limit heap usage
      const batch = await db.query(
        'SELECT * FROM transactions LIMIT $1 OFFSET $2',
        [batchSize, offset]
      );
      
      if (batch.length === 0) {
        hasMore = false;
        break;
      }
      
      await Promise.all(batch.map(txn => this.processSingle(txn)));
      
      offset += batchSize;
      
      // Hint to GC (optional, usually automatic)
      if (global.gc) global.gc();
    }
  }
}

// Object pooling for frequent allocations
class TransactionPool {
  constructor(maxSize = 100) {
    this.pool = [];
    this.maxSize = maxSize;
  }
  
  acquire() {
    return this.pool.pop() || {};
  }
  
  release(obj) {
    if (this.pool.length < this.maxSize) {
      // Clear properties instead of creating new object
      Object.keys(obj).forEach(key => delete obj[key]);
      this.pool.push(obj);
    }
  }
}

const pool = new TransactionPool();

  pool.release(txn);  // Reuse instead of GC
}

🔑 Key Takeaways

  • Stack: Fast, fixed-size, stores primitives and references (LIFO)
  • Heap: Slow, dynamic, stores objects/arrays, managed by GC
  • V8 GC: Generational (Minor GC for young, Major GC for old objects)
  • Memory limits: Default 1.4GB, increase with --max-old-space-size
  • Efficient processing: Use streams, batching, object pooling

✅ DO's

  • Use streams for large datasets
  • Process data in batches to limit memory
  • Monitor memory usage with process.memoryUsage()
  • Use object pooling for frequent allocations
  • Increase heap size for memory-intensive apps

❌ DON'Ts

  • Don't load entire datasets into memory
  • Don't create unnecessary objects in loops
  • Don't hold references to large objects
  • Don't ignore memory warnings and leaks
  • Don't use synchronous operations on large files

// ❌ BAD: Listeners never removed
class LeakyService extends EventEmitter {
  processTransaction(tx) {
    const handler = () => console.log('Done');
    this.on('complete', handler);  // ❌ Never removed!
    this.emit('complete');
  }
}
// After 10,000 calls → 10,000 listeners!

// ✅ GOOD: Remove listeners
class CleanService extends EventEmitter {
  processTransaction(tx) {
    const handler = () => {
      console.log('Done');
      this.removeListener('complete', handler);
    };
    this.once('complete', handler);  // ✅ Auto-removes
    this.emit('complete');
  }
}

// 2. GLOBAL VARIABLE LEAK

// ❌ BAD: Unbounded global array
let allTransactions = [];

function recordTransaction(tx) {
  allTransactions.push(tx);  // ❌ Grows forever
}

// ✅ GOOD: Bounded cache with LRU
const MAX_SIZE = 1000;
let recentTransactions = [];

function recordTransaction(tx) {
  recentTransactions.push(tx);
  if (recentTransactions.length > MAX_SIZE) {
    recentTransactions.shift();  // ✅ Remove oldest
  }
}

// 3. CLOSURE LEAK

// ❌ BAD: Closure captures large object
function processLargeDataset(dataset) {  // 100MB
  return function getFirst() {
    return dataset[0];  // ❌ Keeps entire 100MB
  };
}

// ✅ GOOD: Extract only needed data
function processLargeDataset(dataset) {
  const firstItem = dataset[0];  // ✅ Extract value
  return function getFirst() {
    return firstItem;  // Only captures firstItem
  };
}

// 4. TIMER LEAK

// ❌ BAD: Timer never cleared
function startMonitoring() {
  setInterval(() => {
    checkHealth();
  }, 1000);  // ❌ Runs forever
}

// ✅ GOOD: Clear timers
function startMonitoring() {
  const intervalId = setInterval(() => {
    checkHealth();
  }, 1000);
  
  return () => clearInterval(intervalId);  // ✅ Cleanup
}

const EventEmitter = require('events');

class PaymentProcessor extends EventEmitter {
  constructor() {
    super();
    this.activePayments = new Map();
    this.cache = new LRUCache(1000);  // Bounded cache
    this.timers = new Set();
  }
  
  async processPayment(paymentData) {
    const paymentId = paymentData.id;
    
    // Track active payment
    this.activePayments.set(paymentId, paymentData);
    
    // Set timeout with cleanup
    const timeoutId = setTimeout(() => {
      this.emit('timeout', paymentId);
      this.cleanup(paymentId);
    }, 30000);
    
    this.timers.add(timeoutId);
    
    try {
      // Process payment
      const result = await this.executePayment(paymentData);
      
      // Cache result (bounded)
      this.cache.set(paymentId, result);
      
      return result;
      
    } finally {
      // ✅ Always cleanup
      clearTimeout(timeoutId);
      this.timers.delete(timeoutId);
      this.activePayments.delete(paymentId);
    }
  }
  
  cleanup(paymentId) {
    this.activePayments.delete(paymentId);
  }
  
  shutdown() {
    // Clear all timers
    this.timers.forEach(id => clearTimeout(id));
    this.timers.clear();
    
    // Remove all listeners
    this.removeAllListeners();
    
    // Clear caches
    this.activePayments.clear();
    this.cache.clear();
  }
}

// LRU Cache with bounded size
class LRUCache {
  constructor(maxSize = 100) {
    this.maxSize = maxSize;
    this.cache = new Map();
  }
  
  get(key) {
    if (!this.cache.has(key)) return null;
    
    // Move to end (most recently used)
    const value = this.cache.get(key);
    this.cache.delete(key);
    this.cache.set(key, value);
    return value;
  }
  
  set(key, value) {
    if (this.cache.has(key)) {
      this.cache.delete(key);
    }
    
    this.cache.set(key, value);
    
    // ✅ Evict oldest if over limit
    if (this.cache.size > this.maxSize) {
      const firstKey = this.cache.keys().next().value;
      this.cache.delete(firstKey);
    }
  }
  
  clear() {
    this.cache.clear();
  }
}

// Monitoring memory leaks
function monitorMemory() {
  const initialMemory = process.memoryUsage().heapUsed;
  
  setInterval(() => {
    const currentMemory = process.memoryUsage().heapUsed;
    const diff = currentMemory - initialMemory;
    
    if (diff > 100 * 1024 * 1024) {  // 100MB increase
      console.warn('⚠️  Potential memory leak detected!');
    }
  }, 10000);
}

🔑 Key Takeaways

  • Memory leaks occur when memory is allocated but never freed
  • Common causes: Forgotten event listeners, timers, closures, unbounded caches
  • Detection: Use heap snapshots, Chrome DevTools, process.memoryUsage()
  • Prevention: Remove listeners, clear timers, bound caches, cleanup in finally
  • Tools: --inspect, heapdump, memwatch-next, clinic.js

✅ DO's

  • Use .once() or removeListener() for events
  • Clear all timers when done (clearTimeout/clearInterval)
  • Implement bounded caches (LRU, FIFO)
  • Use WeakMap/WeakSet for objects
  • Monitor memory usage in production

❌ DON'Ts

  • Don't forget to remove event listeners
  • Don't leave timers running indefinitely
  • Don't use unbounded global caches
  • Don't capture large objects in closures
  • Don't ignore increasing memory usage

const transactionsByAccount = new Map();

const balanceCache = new Map();

app.get('/api/account/:id/balance', (req, res) => {
  const accountId = req.params.id;
  
  // Check cache first
  if (balanceCache.has(accountId)) {

      balance: balanceCache.get(accountId),
      cached: true

class TransactionProcessor {
  processDebit(account, amount) {
    // Small focused function (easier to optimize)
    if (account.balance >= amount) {
      account.balance -= amount;
      return true;
    }

    }
  }
}

// Generate JWT token
function generateAccessToken(userId, role) {
  const payload = {
    userId,
    role,
    iat: Date.now()
  };
  
  return jwt.sign(payload, process.env.JWT_SECRET, {
    expiresIn: '15m'  // Short-lived access token
  });
}

// Verify JWT token
function verifyAccessToken(token) {
  try {
    return jwt.verify(token, process.env.JWT_SECRET);
  } catch (error) {
    throw new Error('Invalid token');
  }
}

// Password hashing with bcrypt
async function hashPassword(plainPassword) {
  const saltRounds = 12;  // 2^12 = 4096 iterations (~260ms)
  return await bcrypt.hash(plainPassword, saltRounds);
}

// Verify password
async function verifyPassword(plainPassword, hashedPassword) {
  return await bcrypt.compare(plainPassword, hashedPassword);
}

// Authentication middleware
function authenticateToken(req, res, next) {
  const authHeader = req.headers['authorization'];
  const token = authHeader && authHeader.split(' ')[1];
  
  if (!token) {
    return res.status(401).json({ error: 'Access token required' });
  }
  
  try {
    const decoded = verifyAccessToken(token);
    req.user = decoded;
    next();
  } catch (error) {
    return res.status(403).json({ error: 'Invalid or expired token' });
  }
}

// Role-based authorization
function authorizeRole(...allowedRoles) {
  return (req, res, next) => {
    if (!allowedRoles.includes(req.user.role)) {
      return res.status(403).json({ error: 'Insufficient permissions' });
    }
    next();
  };
}

const crypto = require('crypto');

// AES-256-GCM encryption (industry standard)
class EncryptionService {
  constructor() {
    // 256-bit key (64 hex characters)
    this.algorithm = 'aes-256-gcm';
    this.key = Buffer.from(process.env.ENCRYPTION_KEY, 'hex');
  }
  
  // Encrypt sensitive data
  encrypt(plaintext) {
    // Generate random IV (Initialization Vector)
    const iv = crypto.randomBytes(16);
    
    // Create cipher
    const cipher = crypto.createCipheriv(this.algorithm, this.key, iv);
    
    // Encrypt data
    let encrypted = cipher.update(plaintext, 'utf8', 'hex');
    encrypted += cipher.final('hex');
    
    // Get authentication tag
    const authTag = cipher.getAuthTag();
    
    // Return iv + authTag + encrypted data
    return {
      iv: iv.toString('hex'),
      authTag: authTag.toString('hex'),
      encrypted
    };
  }
  
  // Decrypt encrypted data
  decrypt(encryptedData) {
    const { iv, authTag, encrypted } = encryptedData;
    
    // Create decipher
    const decipher = crypto.createDecipheriv(
      this.algorithm,
      this.key,
      Buffer.from(iv, 'hex')
    );
    
    // Set auth tag
    decipher.setAuthTag(Buffer.from(authTag, 'hex'));
    
    // Decrypt
    let decrypted = decipher.update(encrypted, 'hex', 'utf8');
    decrypted += decipher.final('utf8');
    
    return decrypted;
  }
}

const crypto = require('crypto');

// Banking encryption service for PII data
class BankingEncryption {
  constructor() {
    this.algorithm = 'aes-256-gcm';
    this.masterKey = Buffer.from(process.env.MASTER_KEY, 'hex');
  }
  
  // Encrypt SSN (Social Security Number)
  encryptSSN(ssn) {
    return this.encrypt(ssn);
  }
  
  // Encrypt account number
  encryptAccountNumber(accountNumber) {
    return this.encrypt(accountNumber);
  }
  
  // Encrypt credit card
  encryptCreditCard(cardNumber) {
    return this.encrypt(cardNumber);
  }
  
  encrypt(plaintext) {
    const iv = crypto.randomBytes(16);
    const cipher = crypto.createCipheriv(this.algorithm, this.masterKey, iv);
    
    let encrypted = cipher.update(plaintext, 'utf8', 'base64');
    encrypted += cipher.final('base64');
    
    const authTag = cipher.getAuthTag();
    
    // Combine iv + authTag + encrypted
    const combined = Buffer.concat([
      iv,
      authTag,
      Buffer.from(encrypted, 'base64')
    ]);
    
    return combined.toString('base64');
  }
  
  decrypt(encryptedData) {
    const buffer = Buffer.from(encryptedData, 'base64');
    
    // Extract components
    const iv = buffer.slice(0, 16);
    const authTag = buffer.slice(16, 32);
    const encrypted = buffer.slice(32);
    
    const decipher = crypto.createDecipheriv(this.algorithm, this.masterKey, iv);
    decipher.setAuthTag(authTag);
    
    let decrypted = decipher.update(encrypted);
    decrypted = Buffer.concat([decrypted, decipher.final()]);
    
    return decrypted.toString('utf8');
  }
}

async function executeTransaction(queries) {
  const client = await pool.connect();
  
  try {
    await client.query('BEGIN');

}

    await client.query(

      `INSERT INTO transactions (from_account, to_account, amount, timestamp)
       VALUES ($1, $2, $3, NOW())`,

async function transferWithRetry(from, to, amount, maxRetries = 3) {
